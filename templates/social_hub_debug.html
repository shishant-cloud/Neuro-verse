};

// Wire up buttons
document.getElementById('chatFab')?.addEventListener('click', () => window.chatToggle());
document.getElementById('chatClose')?.addEventListener('click', () => window.chatToggle());

// Load user list from API
async function loadUsers() {
  const res = await fetch('/api/users');
  const { users } = await res.json();
  const list = document.getElementById('chatUserList');
  list.innerHTML = users.map(u => `<div class="chat-user-item" onclick="openConversation(${u.id})" data-uid="${u.id}">${u.username}</div>`).join('');
}

// Load message history
async function openConversation(peerId) {
  activePeerId = peerId;
  document.getElementById('chatInput').disabled = false;
  document.getElementById('chatSend').disabled = false;
  const res = await fetch(`/api/messages/${peerId}`);
  const { messages } = await res.json();
  renderMessages(messages);
}

// Render messages
function renderMessages(msgs) {
  const box = document.getElementById('chatMessages');
  box.innerHTML = msgs.map(m => `<div class="chat-msg ${m.sender_id === CURRENT_USER_ID ? 'mine' : 'theirs'}"><b>${m.sender_username}:</b> ${m.content}</div>`).join('');
  box.scrollTop = box.scrollHeight;
}

// Send message
async function sendMessage() {
  const input = document.getElementById('chatInput');
  const content = input.value.trim();
  if (!content || !activePeerId) return;
  await fetch('/api/messages/send', { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify({ recipient_id: activePeerId, content }) });
  socket.emit('send_message', { sender_id: CURRENT_USER_ID, sender_username: CURRENT_USERNAME, recipient_id: activePeerId, content, created_at: new Date().toISOString() });
  input.value = '';
}

document.getElementById('chatSend')?.addEventListener('click', sendMessage);
document.getElementById('chatInput')?.addEventListener('keydown', e => { if (e.key==='Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); }});

// Real-time: receive messages via socket
socket.on('receive_message', msg => {

Security #5 — init_db.py stores plaintext password (admin account) init_db.py ▶

init_db.py inserts the admin user with password '12345' as plain text. But app.py uses SHA-256 hashing on login, so that admin account will never be able to log in. The passwords must match the same hashing scheme.

❌ init_db.py (broken — plain text)
VALUES ('Admin User', 'admin@neuro.com', '12345', 'Administrator')
✅ init_db.py — use the same hash as app.py
import hashlib

def hash_pw(pw):
  return hashlib.sha256(pw.encode()).hexdigest()

# Then in your INSERT:
VALUES ('Admin User', 'admin@neuro.com', hash_pw('12345'), 'Administrator')

⚠️ Also note: SHA-256 without salt is weak for production. For a real app, use werkzeug.security.generate_password_hash instead.

How Social Hub Works — Architecture
1
Page loads: layout.html renders the chat panel HTML + FAB button hidden by default. It sets CURRENT_USER_ID and loads app.js.
2
Socket connects: app.js creates a Socket.IO connection. The server's on_connect() joins the user to their private room user_<id>.
3
Panel opens: User clicks 💬 → window.chatToggle() → GET /api/users populates the sidebar list.
4
Conversation opens: Clicking a user → GET /api/messages/<peer_id> loads history. Messages are marked read server-side.
5
Send: POST /api/messages/send saves to DB, then socket.emit('send_message') delivers in real-time to user_<recipient_id> room.
6
Receive: Socket event receive_message fires → message is appended live. Unread badge updates via GET /api/unread.
